SDK Integration

Programmatic access to Catacomb intelligence. Four SDKs for every use case.

Proof of Inference SDK

Generate cryptographically verifiable proofs for LLM inferences. SHA-256 + HMAC-SHA256 composite hashing with ledger storage.

proof_example.py 
from proof_of_inference import ProofOfInference, OllamaInferenceClient

# Initialize SDK
poi = ProofOfInference()
ollama = OllamaInferenceClient("http://localhost:11434")

# Generate inference with automatic proof
response = ollama.generate("llama3.1", "Analyze this code for security issues")
proof = poi.generate_proof(
    model="llama3.1",
    prompt="Analyze this code for security issues",
    response=response,
    store_in_ledger=True
)

print(f"Proof ID: {proof.proof_id}")
print(f"Hash: {proof.composite_hash}")
print(f"Signature: {proof.signature}")

# Verify proof independently
is_valid = poi.verify_proof(proof)
print(f"Valid: {is_valid}")

Asset Embeddings SDK

Vector similarity search across software assets. 768-dim embeddings with weighted genome, state, and intervention features.

embeddings_example.py 
from asset_embeddings import AssetEmbeddingGenerator, VectorDatabase, AssetSearchEngine
from asset_abstraction import Asset, AssetGenome, CurrentState

# Initialize components
generator = AssetEmbeddingGenerator(embedding_dim=768)
vector_db = VectorDatabase("asset_embeddings.db")
search_engine = AssetSearchEngine(vector_db, generator)

# Create asset from data
genome = AssetGenome(
    asset_id="langchain/langchain",
    asset_type=AssetType.REPOSITORY,
    primary_language="python",
    stars=50000,
    contributors=200
)

state = CurrentState(
    asset_id="langchain/langchain",
    recognition_score=85.0,
    current_value=75.0
)

asset = Asset(
    asset_id="langchain/langchain",
    asset_type=AssetType.REPOSITORY,
    source=AssetSource.GITHUB,
    genome=genome,
    current_state=state
)

# Index for search
search_engine.index_asset(asset)

# Find similar assets
similar = search_engine.search_similar_assets(
    asset_id="langchain/langchain",
    limit=10,
    asset_type=AssetType.REPOSITORY
)

print(f"Found {len(similar)} similar assets")

ZK Collateral SDK

Zero-Knowledge proofs for inference with economic collateralization. Lock collateral, generate ZK proofs, verify and unlock.

zk_collateral_example.py 
from zk_collateral import ZKCollateralInferenceSDK
from decimal import Decimal

# Initialize SDK
sdk = ZKCollateralInferenceSDK(
    ollama_base_url="http://localhost:11434",
    liquidity_db="liquidity_memory.db"
)

# Create liquidity pool
pool = sdk.liquidity_memory.create_pool(
    pool_id="main_pool",
    collateral_token="USDC",
    initial_liquidity=Decimal("1000000"),
    reward_rate=Decimal("0.05")
)

# Generate inference with ZK proof and collateralize
response, zk_proof, position = sdk.collateralize_inference(
    model="llama3.1",
    prompt="Analyze this repository for intervention opportunities",
    collateral_amount=Decimal("1000"),
    collateral_token="USDC",
    pool_id="main_pool",
    creator="user_123",
    unlock_conditions={"min_accuracy": 0.85}
)

print(f"Response: {response}")
print(f"ZK Proof ID: {zk_proof.proof_id}")
print(f"Position ID: {position.position_id}")
print(f"Collateral Locked: {position.collateral_amount} {position.collateral_token}")

# Verify and unlock collateral
is_valid, message, unlocked_position = sdk.verify_and_unlock(
    position_id=position.position_id,
    model="llama3.1",
    prompt="Analyze this repository for intervention opportunities",
    expected_output=response
)

print(f"Unlocked: {is_valid}")

REST API

Full REST API for programmatic access to all Catacomb features. Stats, interventions, assets, search, and dashboard endpoints.

api_example.py 
import requests

BASE_URL = "https://catacomb-eight.vercel.app/api/v1"

# Get system statistics
response = requests.get(f"{BASE_URL}/stats")
stats = response.json()
print(f"Verified Interventions: {stats['verified_interventions']}")
print(f"Prediction Accuracy: {stats['prediction_accuracy']}%")

# Search for assets
response = requests.get(f"{BASE_URL}/search", params={
    "q": "dependency graph",
    "category": "repository"
})
results = response.json()
print(f"Found {len(results['assets'])} assets")

# Get paginated interventions
response = requests.get(f"{BASE_URL}/interventions", params={
    "page": 1,
    "limit": 10,
    "status": "verified"
})
interventions = response.json()
for intervention in interventions['data']:
    print(f"{intervention['asset_id']}: {intervention['intervention_type']}")

# Get dashboard summary
response = requests.get(f"{BASE_URL}/dashboard/summary")
dashboard = response.json()
print(f"Engineering Alpha: {dashboard['metrics']['engineering_alpha']}")
print(f"Hidden Infrastructure: {dashboard['metrics']['hidden_infrastructure']}")

# Generate proof of inference
response = requests.post(f"{BASE_URL}/inference/proof", json={
    "model": "llama3.1",
    "prompt": "Analyze this code",
    "response": "The code is secure",
    "store_in_ledger": True
})
proof = response.json()
print(f"Proof ID: {proof['proof_id']}")
